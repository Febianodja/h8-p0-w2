2. Tahun Kabisat
    SET "tahun" as any number
        IF 'tahun' % 4 === 0 AND 'tahun' % 100 !== 0 THEN kabisat is true
            ELSE IF 'tahun' % 4 === 0 AND 'tahun' % 100 === 0 AND 'tahun' % 400 === 0 THEN kabisat is true
                ELSE kabisat is false

3. Laundry Day 
    SET 'baju' to 1

4. Periksa Kuku
    SET 'murid' as string that contain 'x' and 'o'
    FOR 'i' EQUAL to 0, i is LESS THAN 'murid' length
        IF index i of 'murid' is 'x' THEN DISPLAY 'puji muridnya'
            ELSE DISPLAY 'hukum muridnya'
